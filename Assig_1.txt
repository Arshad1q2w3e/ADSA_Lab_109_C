#include <iostream>
#include <vector>
using namespace std;

int countZeros(const vector<int>& nums) {
    int n = nums.size();

    if (nums[n - 1] == 1)
        return 0;

    if (nums[0] == 0)
        return n;

    int low = 0, high = n - 1, pos = -1;

    while (low <= high) {
        int mid = (low + high) / 2;

        if (nums[mid] == 0) {
            if (mid == 0 || nums[mid - 1] == 1) {
                pos = mid;
                break;
            } else {
                high = mid - 1;
            }
        } else {
            low = mid + 1;
        }
    }

    if (pos == -1)
        return 0;

    return n - pos;
}

int main() {
    int n;
    cout << "Enter the size of the array: ";
    cin >> n;

    vector<int> nums(n);
    cout << "Enter " << n << " elements (only 1s followed by 0s):\n";
    for (int i = 0; i < n; ++i) {
        cin >> nums[i];
    }

    int zeroCount = countZeros(nums);
    cout << "Number of zeros: " << zeroCount << endl;

    return 0;
}
